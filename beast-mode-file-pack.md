```javascript
  });

  // Your code here
} catch (error) {
  sendErrorAlert(error);
  // Send simplified report or manual fallback
}
```

### Rate Limiting

**Max 20 audits/day:**
```javascript
function checkDailyLimit() {
  const sheet = SpreadsheetApp.getActive().getSheetByName('Audit Log');
  const today = new Date().toDateString();
  const todaysAudits = sheet.getDataRange().getValues()
    .filter(row => row[0].toDateString() === today).length;
  
  if (todaysAudits >= 20) {
    // Queue for tomorrow
    return false;
  }
  return true;
}
```
